<span data-ttu-id="96fe0-127">Im folgenden finden Sie ein Beispiel für eine Anwendungs Konfigurationsdatei.</span><span class="sxs-lookup"><span data-stu-id="96fe0-127">The following is an example of an application configuration file.</span></span> <span data-ttu-id="96fe0-128">Weitere Informationen finden Sie unter Anwendungs Konfigurationsdateien.</span><span class="sxs-lookup"><span data-stu-id="96fe0-128">For more information, see Application Configuration Files.</span></span>

``` syntax
<?xml version="1.0" encoding="UTF-8" standalone="yes"?>
<configuration>
 <windows>
  <assemblyBinding xmlns="urn:schemas-microsoft-com:asm.v1">
   <assemblyIdentity  processorArchitecture="X86" name="Microsoft.Windows.mysampleApp" type="win32" version="1.0.0.0"/>
   <publisherPolicy apply="no"/>                     
   <dependentAssembly>
    <assemblyIdentity type="win32" processorArchitecture="x86" name="Microsoft.Windows.SampleAssembly" publicKeyToken="0000000000000000"/>
    <bindingRedirect oldVersion="2.0.0.0" newVersion="2.0.1.0"/>
   </dependentAssembly>
  </assemblyBinding>
 </windows>
</configuration>
```
