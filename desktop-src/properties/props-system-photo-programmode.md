# <a name="systemphotoprogrammode"></a>System.Photo.ProgramMode

Die Klasse des Programms, das von der Kamera zum Festlegen der Belichtung verwendet wird.

## <a name="windows-10-version-1703-windows-10-version-1607-windows-10-version-1511-windows-10-version-1507-windows-81-windows-8-windows-7"></a>Windows 10, Version 1703, Windows 10, Version 1607, Windows 10, Version 1511, Windows 10, Version 1507, Windows 8.1, Windows 8, Windows 7

```
propertyDescription
   name = System.Photo.ProgramMode
   shellPKey = PKEY_Photo_ProgramMode
   formatID = 6D217F6D-3F6A-4825-B470-5F03CA2FBE9B
   propID = 100
   SearchInfo
      InInvertedIndex = false
      IsColumn = false
   typeInfo
      type = UInt32
      EnumeratedList
         UseValueForDefault = True
         enum
            name = NotDefined
            value = 0
            text = Not defined
            defineToken = PHOTO_PROGRAMMODE_NOTDEFINED
         enum
            name = Manual
            value = 1
            text = Manual
            defineToken = PHOTO_PROGRAMMODE_MANUAL
         enum
            name = Normal
            value = 2
            text = Normal program
            defineToken = PHOTO_PROGRAMMODE_NORMAL
         enum
            name = Aperture
            value = 3
            text = Aperture priority
            defineToken = PHOTO_PROGRAMMODE_APERTURE
         enum
            name = Shutter
            value = 4
            text = Shutter priority
            defineToken = PHOTO_PROGRAMMODE_SHUTTER
         enum
            name = Creative
            value = 5
            text = Creative program
            defineToken = PHOTO_PROGRAMMODE_CREATIVE
         enum
            name = Action
            value = 6
            text = Action program
            defineToken = PHOTO_PROGRAMMODE_ACTION
         enum
            name = Portrait
            value = 7
            text = Portrait
            defineToken = PHOTO_PROGRAMMODE_PORTRAIT
         enum
            name = Landscape
            value = 8
            text = Landscape
            defineToken = PHOTO_PROGRAMMODE_LANDSCAPE
```
