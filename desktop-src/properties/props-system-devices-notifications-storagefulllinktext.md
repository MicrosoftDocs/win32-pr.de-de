propertyDescription
   name = System.Devices.Notifications.StorageFullLinkText
   shellPKey = PKEY_Devices_Notifications_StorageFullLinkText

   propID = 3
   SearchInfo
      InInvertedIndex = false
      IsColumn = false

      IsInnate = true
