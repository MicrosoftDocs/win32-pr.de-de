```
propertyDescription
   name = System.Devices.AepContainer.DialProtocol.InstalledApplications
   shellPKey = PKEY_Devices_AepContainer_DialProtocol_InstalledApplications
   formatID = 6AF55D45-38DB-4495-ACB0-D4728A3B8314
   propID = 6
   SearchInfo
      InInvertedIndex = false

      type = Multivalue String
```
