```
propertyDescription
   name = System.Contact.BusinessAddressPostOfficeBox
   shellPKey = PKEY_Contact_BusinessAddressPostOfficeBox
   formatID = BC4E71CE-17F9-48D5-BEE9-021DF0EA5409
   propID = 100
```
