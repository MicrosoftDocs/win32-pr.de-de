```
propertyDescription
   name = System.Photo.LensManufacturer
   shellPKey = PKEY_Photo_LensManufacturer
   formatID = E6DDCAF7-29C5-4F0A-9A68-D19412EC7090
   propID = 100
   SearchInfo
      InInvertedIndex = false
      IsColumn = false
   typeInfo
      type = String
```
