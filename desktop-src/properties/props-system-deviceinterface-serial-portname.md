Anzeige Name des seriellen Geräts

```
propertyDescription
   name = System.DeviceInterface.Serial.PortName
   shellPKey = PKEY_DeviceInterface_Serial_PortName
   formatID = 4C6BF15C-4C03-4AAC-91F5-64C0F852BCF4
   propID = 4
   SearchInfo

      IsColumn = false
   typeInfo
      type = String
```
